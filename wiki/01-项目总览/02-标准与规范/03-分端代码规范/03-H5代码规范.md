## 三、API 调用（项目 Header 约定）

请求拦截器自动注入 Token 和组织上下文：

```typescript
instance.interceptors.request.use((config) => {
  const token = getAccessToken();
  if (token) {
    config.headers.Authorization = `Bearer ${token}`;
  }

  const orgId = getCurrentOrgId();
  if (orgId) {
    config.headers['X-Organization-ID'] = orgId;
  }

  return config;
});
```

## 四、Token 管理（项目约定）

### 4.1 Access Token：内存中不持久化

```typescript
const tokenInMemory = (() => {
  let token: string | null = null;
  return {
    set: (t: string) => { token = t; },
    get: () => token,
    clear: () => { token = null; },
  };
})();
```

### 4.2 Refresh Token：AES 加密存储

```typescript
function encryptRefreshToken(token: string): string {
  return CryptoJS.AES.encrypt(token, IMPORT.meta.env.VITE_CRYPTO_KEY).toString();
}

function getRefreshToken(): string | null {
  const encrypted = localStorage.getItem('xyfamily_refresh_token');
  if (!encrypted) return null;
  return CryptoJS.AES.decrypt(encrypted, IMPORT.meta.env.VITE_CRYPTO_KEY).toString(CryptoJS.enc.Utf8);
}
```

### 4.3 Token 自动刷新

```typescript
async function handleTokenExpired() {
  if (isRefreshing) return; // 防并发刷新

  isRefreshing = true;
  try {
    const refreshToken = getRefreshToken();
    const response = await api.refresh(refreshToken);
    setAccessToken(response.data.access_token);
    setRefreshToken(response.data.refresh_token);
    retryAllRequests();
  } catch {
    clearTokens();
    navigate('/login');
  } finally {
    isRefreshing = false;
  }
}
```
