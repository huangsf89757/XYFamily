## 三、API 调用封装

### 3.1 Axios 封装

```typescript
import axios from 'axios';

const instance = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL,
  timeout: 30000,
  headers: { 'Content-Type': 'application/json' },
});

// 请求拦截器：注入 Token + 组织上下文
instance.interceptors.request.use((config) => {
  const token = getAccessToken();
  if (token) {
    config.headers.Authorization = `Bearer ${token}`;
  }

  const orgId = getCurrentOrgId();
  if (orgId) {
    config.headers['X-Organization-ID'] = orgId;
  }

  return config;
});

// 响应拦截器：统一错误处理
instance.interceptors.response.use(
  (response) => {
    const { code, message, detail, data } = response.data;
    if (code !== 0) {
      return Promise.reject({ code, message, detail });
    }
    return response.data as APIResponse<any>;
  },
  (error) => {
    const { code, status } = error.response?.data || error.response || {};
    if (status === 401) {
      handleTokenExpired(); // 自动刷新或跳转登录
    }
    return Promise.reject(error);
  }
);
```

### 3.2 Token 自动刷新

```typescript
async function handleTokenExpired() {
  if (isRefreshing) return; // 防止并发刷新

  isRefreshing = true;
  try {
    const refreshToken = getRefreshToken();
    const response = await api.refresh(refreshToken);
    setAccessToken(response.data.access_token);
    setRefreshToken(response.data.refresh_token);
    // 重试所有等待中的请求
    retryAllRequests();
  } catch {
    // Refresh Token 失效 → 清除本地 Token → 跳转登录
    clearTokens();
    navigate('/login');
  } finally {
    isRefreshing = false;
  }
}
```

## 四、本地存储

### 4.1 Token 存储

```typescript
// Access Token：不持久化（页面刷新后需重新登录）
const tokenInMemory = (() => {
  let token: string | null = null;
  return {
    set: (t: string) => { token = t; },
    get: () => token,
    clear: () => { token = null; },
  };
})();

// Refresh Token：localStorage（AES 加密）
function encryptRefreshToken(token: string): string {
  return CryptoJS.AES.encrypt(token, IMPORT.meta.env.VITE_CRYPTO_KEY).toString();
}

function getRefreshToken(): string | null {
  const encrypted = localStorage.getItem('xyfamily_refresh_token');
  if (!encrypted) return null;
  return CryptoJS.AES.decrypt(encrypted, IMPORT.meta.env.VITE_CRYPTO_KEY).toString(CryptoJS.enc.Utf8);
}
```

## 五、状态管理

### 5.1 认证状态

```typescript
import { create } from 'zustand';

interface AuthState {
  user: User | null;
  isAuthenticated: boolean;
  currentOrgId: string | null;
  roles: Record<string, string>; // org_id → role
  permissions: Set<string>;
  fetchUser: () => Promise<void>;
  logout: () => Promise<void>;
  switchOrg: (orgId: string) => void;
}

export const useAuthStore = create<AuthState>((set, get) => ({
  user: null,
  isAuthenticated: false,
  currentOrgId: null,
  roles: {},
  permissions: new Set(),

  fetchUser: async () => {
    const response = await api.getUserProfile();
    set({
      user: response.data,
      isAuthenticated: true,
      roles: response.data.roles || {},
    });
  },

  logout: async () => {
    await api.logout({ refresh_token: getRefreshToken() });
    tokenInMemory.clear();
    localStorage.removeItem('xyfamily_refresh_token');
    set({ user: null, isAuthenticated: false, currentOrgId: null, permissions: new Set() });
    navigate('/login');
  },

  switchOrg: (orgId: string) => {
    set({ currentOrgId: orgId });
    // 切换组织后重新拉取权限
    get().fetchPermissions();
  },
}));
```
