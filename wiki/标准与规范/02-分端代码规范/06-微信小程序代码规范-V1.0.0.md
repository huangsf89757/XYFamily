## 三、API 调用封装

### 3.1 请求封装

```typescript
interface RequestOptions {
  url: string;
  method?: 'GET' | 'POST' | 'PUT' | 'DELETE';
  data?: any;
  skipAuth?: boolean;
}

function request<T>(options: RequestOptions): Promise<T> {
  return new Promise((resolve, reject) => {
    const token = getToken();
    const header: Record<string, string> = {
      'Content-Type': 'application/json',
    };

    if (token && !options.skipAuth) {
      header['Authorization'] = `Bearer ${token}`;
    }

    // 组织上下文
    const orgId = getCurrentOrgId();
    if (orgId) {
      header['X-Organization-ID'] = orgId;
    }

    wx.request({
      url: `${API_BASE_URL}${options.url}`,
      method: options.method || 'GET',
      data: options.data,
      header,
      success: (res) => {
        const { code, message, detail, data } = res.data;
        if (code === 0) {
          resolve(data as T);
        } else if (code >= 101001 && code <= 101009) {
          handleTokenExpired().then(() => {
            resolve(request<T>(options)); // 刷新后重试
          }).catch(() => reject({ code, message }));
        } else {
          reject({ code, message, detail });
        }
      },
      fail: (err) => reject({ code: -1, message: '网络连接异常', ...err }),
    });
  });
}
```

### 3.2 Token 管理

```typescript
const TOKEN_KEY = 'xyfamily_access_token';
const REFRESH_TOKEN_KEY = 'xyfamily_refresh_token';

function getToken(): string | null {
  return wx.getStorageSync(TOKEN_KEY) || null;
}

function setToken(token: string) {
  wx.setStorageSync(TOKEN_KEY, token);
}

function getRefreshToken(): string | null {
  return wx.getStorageSync(REFRESH_TOKEN_KEY) || null;
}

function clearTokens() {
  wx.removeStorageSync(TOKEN_KEY);
  wx.removeStorageSync(REFRESH_TOKEN_KEY);
}

async function handleTokenExpired() {
  const refreshToken = getRefreshToken();
  if (!refreshToken) {
    navigateTo('/pages/login/index');
    return;
  }

  try {
    const res = await request<{ access_token: string; refresh_token: string }>({
      url: '/auth/refresh',
      method: 'POST',
      data: { refresh_token: refreshToken },
      skipAuth: true,
    });
    setToken(res.access_token);
    wx.setStorageSync(REFRESH_TOKEN_KEY, res.refresh_token);
  } catch {
    clearTokens();
    navigateTo('/pages/login/index');
  }
}
```

## 四、页面开发规范

### 4.1 页面结构

```typescript
// Page 配置
Page({
  data: {
    user: null,
    isLoading: false,
  },

  onLoad() {
    this.fetchUserData();
  },

  async fetchUserData() {
    this.setData({ isLoading: true });
    try {
      const user = await request<User>({ url: '/account/profile' });
      this.setData({ user, isLoading: false });
    } catch (err) {
      wx.showToast({ title: getErrorMessage(err.code), icon: 'none' });
      this.setData({ isLoading: false });
    }
  },
});
```

## 六、错误处理

```typescript
function getErrorMessage(code: number): string {
  const map: Record<number, string> = {
    101001: '登录已过期，请重新登录',
    101002: '登录已过期，请重新登录',
    101003: '登录已过期，请重新登录',
    104290: '操作过于频繁，请稍后再试',
    603001: '当前账号无权限执行此操作',
    603002: '当前账号无权限执行此操作',
    805000: '服务异常，请稍后重试',
  };
  return map[code] || '请求失败，请重试';
}

// Toast 统一封装
function showError(code: number) {
  wx.showToast({
    title: getErrorMessage(code),
    icon: 'none',
    duration: 2000,
  });
}
```
