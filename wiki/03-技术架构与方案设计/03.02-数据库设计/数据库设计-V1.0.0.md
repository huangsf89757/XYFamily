## 一、枚举定义

```sql
-- 账号状态
CREATE TYPE account_status AS ENUM ('active', 'deactivating', 'deactivated');

-- 验证码类型
CREATE TYPE vcode_type AS ENUM ('register', 'login', 'reset_password');

-- 验证码状态
CREATE TYPE vcode_status AS ENUM ('pending', 'consumed', 'expired');

-- 邀请类型
CREATE TYPE invitation_target_type AS ENUM ('organization', 'team', 'group');

-- 邀请状态
CREATE TYPE invitation_status AS ENUM ('pending', 'accepted', 'rejected', 'expired');

-- 审计操作类型
CREATE TYPE audit_operation_type AS ENUM (
  -- 认证类
  'auth_register', 'auth_login', 'auth_logout', 'auth_refresh', 'auth_password_reset',
  -- 账号类
  'account_profile_update', 'account_password_change', 'account_deactivate',
  'account_undeactivate', 'account_third_party_bind', 'account_third_party_unbind',
  -- 组织类
  'org_create', 'org_update', 'org_member_invite', 'org_member_remove',
  'org_role_assign', 'org_role_downgrade', 'org_team_create',
  -- 团队类
  'team_create', 'team_update', 'team_member_invite', 'team_member_remove',
  'team_role_assign', 'team_role_downgrade', 'team_group_create', 'team_archive',
  -- 小组类
  'group_create', 'group_update', 'group_delete', 'group_member_invite',
  'group_member_remove', 'group_role_assign', 'group_role_downgrade',
  -- 管理类
  'admin_config_update', 'admin_force_downgrade'
);
```

## 二、账号层

### 2.1 `accounts` — 账号聚合根

```sql
CREATE TABLE accounts (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  account_id UUID NOT NULL DEFAULT gen_random_uuid(),
  phone VARCHAR(20),
  email VARCHAR(255),
  username VARCHAR(50),
  password_hash VARCHAR(255),
  nickname VARCHAR(100),
  avatar_url TEXT,
  status account_status NOT NULL DEFAULT 'active',
  deactivated_at TIMESTAMP,
  expires_at TIMESTAMP,
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  last_login_at TIMESTAMP,

  CONSTRAINT accounts_has_identifier CHECK (phone IS NOT NULL OR email IS NOT NULL)
);

CREATE UNIQUE INDEX idx_accounts_phone ON accounts(phone) WHERE phone IS NOT NULL;
CREATE UNIQUE INDEX idx_accounts_email ON accounts(email) WHERE email IS NOT NULL;
CREATE UNIQUE INDEX idx_accounts_username ON accounts(username) WHERE username IS NOT NULL;
CREATE UNIQUE INDEX idx_accounts_account_id ON accounts(account_id);
CREATE INDEX idx_accounts_status ON accounts(status);
CREATE INDEX idx_accounts_expires_at ON accounts(expires_at) WHERE expires_at IS NOT NULL;
```

### 2.2 `verification_codes` — 验证码

```sql
CREATE TABLE verification_codes (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  target VARCHAR(255) NOT NULL,
  code VARCHAR(6) NOT NULL,
  type vcode_type NOT NULL,
  status vcode_status NOT NULL DEFAULT 'pending',
  expires_at TIMESTAMP NOT NULL,
  attempt_count INT NOT NULL DEFAULT 0,
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE UNIQUE INDEX idx_vcode_target_type_pending ON verification_codes(target, type)
  WHERE status = 'pending';
CREATE INDEX idx_vcode_status ON verification_codes(status);
CREATE INDEX idx_vcode_expires_at ON verification_codes(expires_at);
```

### 2.3 `invitations` — 邀请记录

```sql
CREATE TABLE invitations (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  inviter_account_id UUID NOT NULL REFERENCES accounts(id),
  invitee_identifier VARCHAR(255) NOT NULL,
  invitee_account_id UUID REFERENCES accounts(id),
  target_type invitation_target_type NOT NULL,
  target_id UUID NOT NULL,
  target_role VARCHAR(20) NOT NULL DEFAULT 'RegularMember',
  status invitation_status NOT NULL DEFAULT 'pending',
  expires_at TIMESTAMP NOT NULL,
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_invitations_invitee_account_id ON invitations(invitee_account_id);
CREATE INDEX idx_invitations_status ON invitations(status);
CREATE INDEX idx_invitations_expires_at ON invitations(expires_at);
```

### 2.4 `sessions` — 会话（Refresh Token 有状态存储）

```sql
CREATE TABLE sessions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  account_id UUID NOT NULL REFERENCES accounts(id),
  refresh_token VARCHAR(255) NOT NULL,
  device_info VARCHAR(255),
  ip_address VARCHAR(45),
  expires_at TIMESTAMP NOT NULL,
  revoked BOOLEAN NOT NULL DEFAULT FALSE,
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE UNIQUE INDEX idx_sessions_refresh_token ON sessions(refresh_token);
CREATE INDEX idx_sessions_account_id ON sessions(account_id);
CREATE INDEX idx_sessions_expires_at ON sessions(expires_at);
```

## 三、组织架构层

### 3.1 `organizations` — 组织

```sql
CREATE TABLE organizations (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name VARCHAR(100) NOT NULL,
  logo TEXT,
  description TEXT,
  status VARCHAR(20) NOT NULL DEFAULT 'active',
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  deleted_at TIMESTAMP
);

CREATE UNIQUE INDEX idx_organizations_name ON organizations(name)
  WHERE deleted_at IS NULL;
```

### 3.2 `organization_members` — 组织成员

```sql
CREATE TABLE organization_members (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  organization_id UUID NOT NULL REFERENCES organizations(id),
  account_id UUID NOT NULL REFERENCES accounts(id),
  role VARCHAR(20) NOT NULL DEFAULT 'RegularMember',
  status VARCHAR(20) NOT NULL DEFAULT 'active',
  joined_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,

  CONSTRAINT org_members_role_check CHECK (
    role IN ('organization_core_admin', 'organization_ordinary_admin', 'RegularMember')
  ),
  UNIQUE(account_id, organization_id)
);

CREATE INDEX idx_org_members_org_id ON organization_members(organization_id);
CREATE INDEX idx_org_members_account_id ON organization_members(account_id);
```

### 3.3 `teams` — 团队

```sql
CREATE TABLE teams (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  organization_id UUID NOT NULL REFERENCES organizations(id),
  name VARCHAR(100) NOT NULL,
  description TEXT,
  status VARCHAR(20) NOT NULL DEFAULT 'active',
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  deleted_at TIMESTAMP
);

CREATE UNIQUE INDEX idx_teams_name_org ON teams(name, organization_id)
  WHERE deleted_at IS NULL;
```

### 3.4 `team_members` — 团队成员

```sql
CREATE TABLE team_members (
```

### 4.1 `roles` — 角色

```sql
CREATE TABLE roles (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name VARCHAR(50) NOT NULL,
  code VARCHAR(25) NOT NULL,
  level INT NOT NULL,
  description TEXT,
  is_system BOOLEAN NOT NULL DEFAULT TRUE,
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,

  CONSTRAINT roles_code_unique UNIQUE(code),
  CONSTRAINT roles_level_check CHECK (
    level IN (0, 1, 2, 3, 4, 5)
  )
);
```

### 4.3 `role_permissions` — 角色权限映射

```sql
CREATE TABLE role_permissions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  role_id UUID NOT NULL REFERENCES roles(id),
  permission_id UUID NOT NULL REFERENCES permissions(id),
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  UNIQUE(role_id, permission_id)
);
```

**预初始化映射规则：**
- Public（L0）：auth.register, auth.login, auth.reset_password
- RegularMember（L1）：account 全部 + 基础 read
- group_ordinary_admin（L2）：RegularMember + group 基础管理（成员邀请/移除/列表、资源读取）
- group_core_admin（L3）：group_ordinary_admin + group 全部（含修改/删除小组、分配/降级角色）
- team_ordinary_admin（L4）：group_core_admin + team 基础管理（成员邀请/移除/列表、资源读取）
- team_core_admin（L5）：team_ordinary_admin + team 全部（含修改/归档团队、创建小组、分配/降级角色）
- organization_ordinary_admin（L6）：team_core_admin + 组织基础管理（成员邀请/移除/列表、资源读取）
- organization_core_admin（L7）：organization_ordinary_admin + org 全部（含修改组织、创建团队、分配/降级角色、查看审计）
- SuperAdmin（L8）：全部 45 个权限点 + admin 全部

> 映射种子来源：上述角色×权限的完整映射矩阵见 `02.02-产品PRD/07-权限管理模块/权限管理模块-V1.0.0.md` 的「权限点矩阵」章节；数据库 seeds 以此矩阵为准，避免两份数据分叉。

## 五、审计层

### 5.1 `login_audit_logs` — 登录审计日志

```sql
CREATE TABLE login_audit_logs (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  account_id UUID REFERENCES accounts(id),
  login_method VARCHAR(20) NOT NULL,
  login_result VARCHAR(20) NOT NULL,
  failure_reason VARCHAR(255),
  ip_address VARCHAR(45),
  user_agent TEXT,
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_login_audit_account_id ON login_audit_logs(account_id);
CREATE INDEX idx_login_audit_created_at ON login_audit_logs(created_at);
CREATE INDEX idx_login_audit_result ON login_audit_logs(login_result);
```

**`login_method` 枚举值**：`phone_password`, `email_password`, `username_password`, `phone_code`, `email_code`, `user_id_password`（**不含 wechat**，延后）

### 5.2 `operation_audit_logs` — 操作审计日志

```sql
CREATE TABLE operation_audit_logs (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  account_id UUID REFERENCES accounts(id),
  operation_type audit_operation_type NOT NULL,
  target_type VARCHAR(20),
  target_id UUID,
  org_id UUID REFERENCES organizations(id),
  details JSONB,
  ip_address VARCHAR(45),
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_op_audit_account_id ON operation_audit_logs(account_id);
CREATE INDEX idx_op_audit_org_id ON operation_audit_logs(org_id);
CREATE INDEX idx_op_audit_operation_type ON operation_audit_logs(operation_type);
CREATE INDEX idx_op_audit_created_at ON operation_audit_logs(created_at);
```

**审计日志写入方式**：通过 Redis Stream 异步写入（见中间件专项方案），保证不阻塞主流程且不丢失数据。
